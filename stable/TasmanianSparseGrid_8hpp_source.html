1843  DomainInsideSignature getDomainInside() const{
1844  auto rule = getRule();
1845  if ((rule == TasGrid::rule_gausshermite) || (rule == TasGrid::rule_gausshermiteodd)){ // unbounded domain
1846  return [](std::vector<double> const &)->bool{ return true; };
1847  }else if ((rule == TasGrid::rule_gausslaguerre) || (rule == TasGrid::rule_gausslaguerreodd)){ // bounded from below
1848  if (domain_transform_a.empty()){
1849  return [](std::vector<double> const &x)->bool{
1850  for(auto const &v : x) if (v < 0.0) return false;
1851  return true;
1852  };
1853  }else{
1854  size_t dims = (size_t) getNumDimensions();
1855  return [=](std::vector<double> const &x)->bool{
1856  for(size_t i=0; i<dims; i++) if (x[i] < domain_transform_a[i]) return false;
1857  return true;
1858  };
1859  }
1860  }else{
1861  if (domain_transform_a.empty()){
1862  if (isFourier()){
1863  return [](std::vector<double> const &x)->bool{
1864  for(auto const &v : x) if ((v < 0.0) || (v > 1.0)) return false;
1865  return true;
1866  };
1867  }else{
1868  return [](std::vector<double> const &x)->bool{
1869  for(auto const &v : x) if ((v < -1.0) || (v > 1.0)) return false;
1870  return true;
1871  };
1872  }
1873  }else{
1874  size_t dims = (size_t) getNumDimensions();
1875  return [=](std::vector<double> const &x)->bool{
1876  for(size_t i=0; i<dims; i++)
1877  if ((x[i] < domain_transform_a[i]) || (x[i] > domain_transform_b[i])) return false;
1878  return true;
1879  };
1880  }
1881  }
1882  }
