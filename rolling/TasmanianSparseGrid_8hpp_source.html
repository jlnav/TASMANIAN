1878  DomainInsideSignature getDomainInside() const{
1879  auto rule = getRule();
1880  if ((rule == TasGrid::rule_gausshermite) || (rule == TasGrid::rule_gausshermiteodd)){ // unbounded domain
1881  return [](std::vector<double> const &)->bool{ return true; };
1882  }else if ((rule == TasGrid::rule_gausslaguerre) || (rule == TasGrid::rule_gausslaguerreodd)){ // bounded from below
1883  if (domain_transform_a.empty()){
1884  return [](std::vector<double> const &x)->bool{
1885  for(auto const &v : x) if (v < 0.0) return false;
1886  return true;
1887  };
1888  }else{
1889  size_t dims = (size_t) getNumDimensions();
1890  return [=](std::vector<double> const &x)->bool{
1891  for(size_t i=0; i<dims; i++) if (x[i] < domain_transform_a[i]) return false;
1892  return true;
1893  };
1894  }
1895  }else{
1896  if (domain_transform_a.empty()){
1897  if (isFourier()){
1898  return [](std::vector<double> const &x)->bool{
1899  for(auto const &v : x) if ((v < 0.0) || (v > 1.0)) return false;
1900  return true;
1901  };
1902  }else{
1903  return [](std::vector<double> const &x)->bool{
1904  for(auto const &v : x) if ((v < -1.0) || (v > 1.0)) return false;
1905  return true;
1906  };
1907  }
1908  }else{
1909  size_t dims = (size_t) getNumDimensions();
1910  return [=](std::vector<double> const &x)->bool{
1911  for(size_t i=0; i<dims; i++)
1912  if ((x[i] < domain_transform_a[i]) || (x[i] > domain_transform_b[i])) return false;
1913  return true;
1914  };
1915  }
1916  }
