626  ChangeType favorSparse(bool favor){
627  AlgorithmPreference new_preference = [=]()->AlgorithmPreference{
628  if (favor){
629  return (algorithm_select == algorithm_dense) ? algorithm_autoselect : algorithm_sparse;

637  }else{

639  }

661  ChangeType testEnable(TypeAcceleration acc, int new_gpu_id) const{
662  TypeAcceleration effective_acc = AccelerationMeta::getAvailableFallback(acc);
663  if (AccelerationMeta::isAccTypeGPU(effective_acc) and ((new_gpu_id < 0 or new_gpu_id >= AccelerationMeta::getNumGpuDevices())))
664  throw std::runtime_error("Invalid GPU device ID, see ./tasgrid -v for list of detected devices.");
665  ChangeType mode_change = (effective_acc == mode) ? change_none : change_cpu_blas;
666  ChangeType device_change = (device == new_gpu_id) ? change_none : change_gpu_device;
667 
668  if (on_gpu()){
669  return (AccelerationMeta::isAccTypeGPU(effective_acc)) ? device_change : change_gpu_device;
670  }else{
671  return (AccelerationMeta::isAccTypeGPU(effective_acc)) ? change_gpu_enabled : mode_change;
672  }
673  }

676  void enable(TypeAcceleration acc, int new_gpu_id){
677  // get the effective new acceleration mode (use the fallback if acc is not enabled)
678  TypeAcceleration effective_acc = AccelerationMeta::getAvailableFallback(acc);
679  // if switching to a GPU mode, check if the device id is valid
680  if (AccelerationMeta::isAccTypeGPU(effective_acc) and ((new_gpu_id < 0 or new_gpu_id >= AccelerationMeta::getNumGpuDevices())))
681  throw std::runtime_error("Invalid GPU device ID, see ./tasgrid -v for list of detected devices.");
682 
683  if (AccelerationMeta::isAccTypeGPU(effective_acc)){
684  // if the new mode is GPU-based, make an engine or reset the engine if the device has changed
685  // if the engine exists and the device is not changed, then keep the existing engine
686  if (!engine or new_gpu_id != device)
687  engine = Utils::make_unique<GpuEngine>();
688  }else{
689  engine.reset();
690  }
691 
692  // assign the new values for the mode and device
693  mode = effective_acc;
694  device = new_gpu_id;
695  }
697  void setDevice() const{ AccelerationMeta::setDefaultGpuDevice(device); }
699  operator GpuEngine* () const{ return engine.get(); }
701  bool on_gpu() const{ return !!engine; }
702 };
