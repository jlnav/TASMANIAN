89  static void cacheLevel(int level, double x, const OneDimensionalWrapper &rule, T *cc){
90  const double *nodes = rule.getNodes(level);
91  const double *coeff = rule.getCoefficients(level);
92  int num_points = rule.getNumPoints(level);
93 
94  cc[0] = 1.0;
95  T c = 1.0;
96  for(int j=0; j<num_points-1; j++){
97  c *= (x - nodes[j]);
98  cc[j+1] = c;
99  }
100  c = (rule.getType() == rule_clenshawcurtis0) ? (x * x - 1.0) : 1.0;
101  cc[num_points-1] *= c * coeff[num_points-1];
102  for(int j=num_points-2; j>=0; j--){
103  c *= (x - nodes[j+1]);
104  cc[j] *= c * coeff[j];
105  }
106  }
107

110  return cache[dimension][offsets[level] + local];

154  static void cacheDerivativeLevel(int level, double x, const OneDimensionalWrapper &rule, T *vals, T *cc){
155  // Initialize.
156  const double *nodes = rule.getNodes(level);
157  const double *coeff = rule.getCoefficients(level);
158  int num_points = rule.getNumPoints(level);
159 
160  // Check to see if x is a root of the Lagrange polynomial.
161  int match_idx = -1;
162  for (int j=0; j<num_points; j++) {
163  if (std::fabs(x - nodes[j]) <= Maths::num_tol) {
164  match_idx = j;
165  break;
166  }
167  }
168 
169  // The Clenshaw-Curtis-Zero rule multiplies (x - 1.0)(x + 1.0) to the usual polynomial. The gradient of resulting
170  // polynomial should hence be modified if x is 1.0 or -1.0.
171  bool clenshaw_bdy_point = rule.getType() == rule_clenshawcurtis0 and std::fabs(x * x - 1.0) <= Maths::num_tol;
172 
173  if (match_idx >= 0 or clenshaw_bdy_point) {
174  // Gradient evaluation at one of roots of the Lagrange polynomial. Must be built from scratch.
175  cc[0] = 1.0;
176  T c = 1.0;
177  T inv_sum = (1.0 * (0 != match_idx) + 0.0 * (0 == match_idx)) /
178  ((x - nodes[0]) * (0 != match_idx) + 1.0 * (0 == match_idx));
179  for (int j=0; j<num_points-1; j++) {
180  c *= (x - nodes[j]) * (j != match_idx) + 1.0 * (j == match_idx);
181  cc[j+1] = c;
182  inv_sum += (1.0 * (j+1 != match_idx) + 0.0 * (j+1 == match_idx)) /
183  ((x - nodes[j+1]) * (j+1 != match_idx) + 1.0 * (j+1 == match_idx));
184  }
185  c = 1.0;
186  if (rule.getType() == rule_clenshawcurtis0)
187  c = clenshaw_bdy_point ? 2.0 * x : (x - 1.0) * (x + 1.0);
188  cc[num_points-1] *= c * coeff[num_points-1];
189  for(int j=num_points-2; j>=0; j--) {
190  c *= (x - nodes[j+1]) * (j+1 != match_idx) + 1.0 * (j+1 == match_idx);
191  cc[j] *= c * coeff[j];
192  }
193  // cc[match_idx] is a special case due to the definition of the Lagrange polynomial.
194  if (!clenshaw_bdy_point)
195  cc[match_idx] = inv_sum;
196  } else {
197  // Gradient evaluation at all other points. Can be built from the cached Lagrange values.
198  T inv_sum = 0.0;
199  for (int i=0; i<num_points; i++)
200  inv_sum += 1.0 / (x - nodes[i]);
201  if (rule.getType() == rule_clenshawcurtis0)
202  inv_sum += 1.0 / (x - 1.0) + 1.0 / (x + 1.0);
203  for (int j=0; j<num_points; j++)
204  cc[j] = vals[j] * (inv_sum - 1.0 / (x - nodes[j]));
205  }
206  }
207 
209  T getLagrangeDerivative(int dimension, int level, int local) const {
210  return cacheDerivatives[dimension][offsets[level] + local];
211  }
212 
213 protected:
214  std::vector<std::vector<T>> cacheDerivatives;
215  // Needed for proper inheritance.
216  using CacheLagrange<T>::offsets;
217  using CacheLagrange<T>::cache;
218 };
