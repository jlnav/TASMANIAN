97 inline ObjectiveFunction makeObjectiveFunction(const int num_dimensions, const ObjectiveFunctionSingle f_single) {
98  return [=](const std::vector<double> &x_values, std::vector<double> &fval_values)->void {
99  int num_points = x_values.size() / num_dimensions;
100  std::vector<double> x(num_dimensions);
101  for (int i=0; i<num_points; i++) {
102  std::copy_n(x_values.begin() + i * num_dimensions, num_dimensions, x.begin());
103  fval_values[i] = f_single(x);
104  }
105  };
